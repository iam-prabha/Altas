TypeScript Syntax Reference

Quick reference for everyday application TypeScript

Primitive Types

let s: string = "hello";
let n: number = 42;
let b: boolean = true;
let n: null = null;
let u: undefined = undefined;

Type Inference

let name = "Alice";        // inferred: string
let age = 30;              // inferred: number
const roles = ["admin"];   // inferred: string[]

Object Types (Interfaces & Type Aliases)

// Prefer interface for object shapes in application code
interface User {
  id: number;
  name: string;
  email?: string;          // optional property
}

// Use type alias for unions, primitives, tuples
type Status = "active" | "inactive";
type Callback = (result: string) => void;

Function Types

// Annotated function
function add(a: number, b: number): number {
  return a + b;
}

// Arrow function with type
const multiply = (a: number, b: number): number => a * b;

// Function type as a signature
type MathOp = (a: number, b: number) => number;
const divide: MathOp = (a, b) => a / b;

Arrays & Tuples

const nums: number[] = [1, 2, 3];
const names: Array<string> = ["a", "b"];

// Tuple — fixed length, known types
const pair: [string, number] = ["age", 30];

Union & Intersection Types

// Union — value can be one of several types
let id: string | number;
id = "abc";  // ok
id = 123;    // ok

// Intersection — must satisfy all types
type Named = { name: string };
type Aged = { age: number };
type Person = Named & Aged;
// { name: string; age: number; }

Type Narrowing

function formatId(id: string | number): string {
  if (typeof id === "string") {
    return id.toUpperCase();  // narrowed to string
  }
  return id.toFixed(0);       // narrowed to number
}

Generics (Quick Look)

function first<T>(arr: T[]): T | undefined {
  return arr[0];
}

const num = first([1, 2, 3]);     // inferred: number
const str = first(["a", "b"]);    // inferred: string

Utility Types

interface User { id: number; name: string; email: string; }

type PartialUser = Partial<User>;   // all fields optional
type UserName = Pick<User, "name">; // { name: string }
type WithoutEmail = Omit<User, "email">; // { id: number; name: string }
